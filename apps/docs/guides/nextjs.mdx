## Setup

```bash
npm install -D the-prompting-library
```

```json package.json
{
  "scripts": {
    "dev": "tpl watch & next dev",
    "build": "tpl generate && next build"
  }
}
```

## Create a prompt next to your route

```markdown src/app/api/chat/system.tpl.md
---
description: System prompt for the chat API
---

You are a helpful assistant for {{companyName}}.
Respond in {{language|English}}. Keep responses concise.
```

## Use in a route handler

```typescript src/app/api/chat/route.ts
import { prompts } from "@/lib/tpl";

export async function POST(request: Request) {
  const { messages } = await request.json();

  const systemPrompt = prompts.system({
    companyName: "Acme Corp",
    // language is optional — defaults to "English"
  });

  // Use systemPrompt with your preferred LLM SDK
}
```

## TypeScript path alias

If you use `@/` path aliases, ensure `lib/` is reachable:

```json tsconfig.json
{
  "compilerOptions": {
    "paths": {
      "@/*": ["./*"]
    }
  }
}
```

## Gitignore (optional)

```gitignore
lib/tpl/
```

Or commit it — both are valid. Committing means PR diffs show exactly what changed in the generated output.
